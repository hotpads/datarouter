CREATE TABLE `Inquiry` (
`id` bigint(20) NOT NULL AUTO_INCREMENT,
`date` int(11) DEFAULT NULL,
`day` int(11) DEFAULT NULL,
`email` varchar(255) DEFAULT NULL,
`feed` varchar(255) DEFAULT NULL,
`feedListingId` varchar(255) DEFAULT NULL,
`feedCompanyId` varchar(100) DEFAULT NULL,
`hour` int(11) DEFAULT NULL,
`listingType` varchar(255) DEFAULT NULL,
`message` mediumtext,
`minute` int(11) DEFAULT NULL,
`month` int(11) DEFAULT NULL,
`movingDate` varchar(255) DEFAULT NULL,
`name` varchar(255) DEFAULT NULL,
`numBathroomsSought` varchar(255) DEFAULT NULL,
`numBedroomsSought` varchar(255) DEFAULT NULL,
`phone` varchar(255) DEFAULT NULL,
`second` int(11) DEFAULT NULL,
`session` varchar(255) DEFAULT NULL,
`state` varchar(255) DEFAULT NULL,
`timestamp` bigint(20) DEFAULT NULL,
`year` int(11) DEFAULT NULL,
`zip` varchar(255) DEFAULT NULL,
`delivered` bit(1) DEFAULT NULL,
`ipAddress` varchar(255) DEFAULT NULL,
`contactEmail` varchar(255) DEFAULT NULL,
`awaitingPayment` bit(1) DEFAULT NULL,
`price` double DEFAULT NULL,
`valid` bit(1) DEFAULT NULL,
`paid` bit(1) DEFAULT NULL,
`invoice` bit(1) DEFAULT NULL,

`expired` bit(1) DEFAULT NULL,
`ref` varchar(100) DEFAULT NULL,
`remoteId` varchar(255) DEFAULT NULL,
`emailOpened` datetime DEFAULT NULL,
`ccEmailOpened` datetime DEFAULT NULL,
`userToken` varchar(255) DEFAULT NULL,
PRIMARY KEY (`id`),
KEY `index_yyyymmddhhmmss` (`year`,`month`,`date`,`hour`,`minute`,`second`),
KEY `index_awaitingPayment` (`awaitingPayment`),
KEY `index_price_invoice_valid_paid_expired` (`price`,`invoice`,`valid`,`paid`,`expired`),
KEY `index_feed_feedListingId` (`feed`,`feedListingId`),
KEY `index_ipAddress` (`ipAddress`)
) ENGINE=InnoDB AUTO_INCREMENT=6853302 DEFAULT CHARSET=latin1
